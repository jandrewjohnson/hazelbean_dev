class TestMixedHierarchyConstruction:
    """Tests for iterators with child tasks, and complex nesting patterns."""

    def test_iterator_with_child_tasks_hierarchy(self, isolated_project):
        """Test iterator as parent with multiple child tasks."""

        def setup_iterator(p):
            """Iterator function that prepares scenarios for child tasks."""
            p.iterator_replacements = {
                'scenario_name': ['alpha', 'beta'],
                'cur_dir_parent_dir': [
                    os.path.join(p.cur_dir, 'alpha'), 
                    os.path.join(p.cur_dir, 'beta')
                ]
            }
            # Create directories for each scenario
            for dir_path in p.iterator_replacements['cur_dir_parent_dir']:
                os.makedirs(dir_path, exist_ok=True)

        def child_task_a(p):
            """First child task that processes scenario data."""
            output_path = os.path.join(p.cur_dir, f'output_a_{p.scenario_name}.txt')
            with open(output_path, 'w') as f:
                f.write(f"Child task A processing scenario: {p.scenario_name}")

        def child_task_b(p):  
            """Second child task that processes scenario data."""
            output_path = os.path.join(p.cur_dir, f'output_b_{p.scenario_name}.txt')
            with open(output_path, 'w') as f:
                f.write(f"Child task B processing scenario: {p.scenario_name}")

        # Create mixed hierarchy: iterator with child tasks
        iterator = isolated_project.add_iterator(setup_iterator, run_in_parallel=False)
        task_a = isolated_project.add_task(child_task_a, parent=iterator)
        task_b = isolated_project.add_task(child_task_b, parent=iterator)

        # Validate hierarchy structure
        assert iterator in isolated_project.task_tree.children
        assert task_a.parent == iterator
        assert task_b.parent == iterator

        # Validate performance within acceptable bounds
        assert execution_time < 5.0  # Should complete within 5 seconds
        assert len(isolated_project.task_names_defined) == 100

    def test_iterator_parallel_performance(self, isolated_project):
        """Test iterator performance with parallel processing."""
        def performance_iterator(p):
            p.iterator_replacements = {
                'scenario_id': list(range(10)),  # 10 scenarios
                'cur_dir_parent_dir': [
                    os.path.join(p.cur_dir, f'scenario_{i}') for i in range(10)
                ]
            }
            for dir_path in p.iterator_replacements['cur_dir_parent_dir']:
                os.makedirs(dir_path, exist_ok=True)

        start_time = time.perf_counter()
        iterator = isolated_project.add_iterator(
            performance_iterator, 
            run_in_parallel=True,
            max_workers=4
        )
        end_time = time.perf_counter()

        setup_time = end_time - start_time
        assert setup_time < 2.0  # Iterator setup should be fast
        assert len(iterator.iterator_replacements['scenario_id']) == 10

    Critical Pattern: Proper cleanup prevents test contamination and memory leaks.

def anytree_node_tracker():
    """Track anytree nodes for memory leak detection."""
    import gc
    import anytree

    # Count nodes before test
    gc.collect()
    initial_nodes = [obj for obj in gc.get_objects() 
                    if isinstance(obj, anytree.NodeMixin)]
    initial_count = len(initial_nodes)

    yield

    # Clean up and validate after test
    gc.collect()
    final_nodes = [obj for obj in gc.get_objects() 
                  if isinstance(obj, anytree.NodeMixin)]
    final_count = len(final_nodes)

    node_difference = final_count - initial_count
    if node_difference > 5:  # Allow small retention for complex tests
        pytest.fail(f"Significant memory retention: {node_difference} nodes retained")
    elif node_difference > 0:
        print(f"Warning: Minor memory retention detected: {node_difference} nodes (acceptable for checkpoint)")

@pytest.fixture  
def isolated_project():
    """Create isolated ProjectFlow instance with automatic cleanup."""
    import tempfile
    import shutil
    import hazelbean as hb

    # Create temporary directory
    temp_dir = tempfile.mkdtemp(prefix='test_projectflow_')

    try:
        # Initialize ProjectFlow with isolated directory
        project = hb.ProjectFlow(temp_dir)
        yield project

    finally:
        # Cleanup: remove temporary files and clear references
        if os.path.exists(temp_dir):
            shutil.rmtree(temp_dir, ignore_errors=True)
        del project

    Automated Performance Tracking¶

    The system includes sophisticated performance monitoring:

    # Performance Baseline Example
class PerformanceBaseline:
    """Manages performance baselines for regression detection."""

    def establish_baseline(self, test_name, execution_time, metadata):
        """Establish new performance baseline."""
        baseline_data = {
            'test_name': test_name,
            'execution_time': execution_time,
            'timestamp': datetime.now().isoformat(),
            'metadata': metadata,
            'confidence_interval': self.calculate_confidence_interval([execution_time])
        }
        self.save_baseline(baseline_data)

    def check_regression(self, test_name, current_time, threshold=10.0):
        """Check for performance regression against baseline."""
        baseline = self.load_baseline(test_name)
        if not baseline:
            return {'status': 'no_baseline', 'action': 'establish_baseline'}

        regression_percentage = ((current_time - baseline['execution_time']) / 
                               baseline['execution_time']) * 100

        if regression_percentage > threshold:
            return {
                'status': 'regression_detected',
                'regression_percent': regression_percentage,
                'action': 'investigation_required'
            }
        else:
            return {
                'status': 'performance_acceptable',
                'regression_percent': regression_percentage,
                'action': 'continue'
            }

    Common Test Failures & Solutions¶

    1. Memory Leak Detection Failures
    2. Cause: anytree nodes not properly cleaned up
    3. Solution: Use clean_task_tree() fixture and explicit cleanup

    4. Prevention: Always use isolated_project fixture

    5. Test Contamination Issues

    6. Cause: Tests affecting each other through shared state
    7. Solution: Ensure each test uses fresh ProjectFlow instance

    8. Prevention: Use proper test isolation patterns

    9. Performance Regression Failures

    10. Cause: Code changes affecting execution speed
    11. Solution: Review changes for performance impact

    12. Prevention: Include performance tests in development workflow

    13. Error Handling Test Failures

    14. Cause: Bugs in hazelbean ProjectFlow error handling
    15. Action: Document as known issue, not test failure
    16. Resolution: Update hazelbean library error handling
